🛡️ 内存安全系统

三级指针体系

safe {
    // 独占所有权指针
    own<vector<int>> data = own_new<vector<int>>(100);
    
    // 安全借用
    borrow <mut int> item = data[0];
    
    // 跨域转换
    legacy_ptr<int> raw = data.leak_unsafe();
}

混合垃圾回收

gc --mode=generational {
    Gc<Connection> conn = gc_new<Connection>();
    
    // 混合内存管理
    own<Socket> sock = own_new<Socket>();
    gc_adopt(sock.release());
}

⚡ 并发与并行

结构化并发

async scope {
    let task1 = co_spawn fetch_data();
    Chan<int> pipe(10);
    
    par {
        pipe.send(co_await task1);
        process(pipe.recv().unwrap());
    }
    
    -> merge(task1) {
        log!("任务完成");
    }
}

无锁编程

LockFreeMap<int, string> cache;

atomic {
    cache.insert(42, "answer");
    if (let Some(val) = cache.get(42)) {
        process(val);
    }
}

    Button btn;
    btn.clicked += [](int x, int y) {
        println!("点击坐标: ({}, {})", x, y);
    };
    
    async {
        co_await btn.listen();
    };
}
